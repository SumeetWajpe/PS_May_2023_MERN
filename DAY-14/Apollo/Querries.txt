
==== Get a course by Id (variables) =====
query GetCourseById($courseId: ID!) {
  course(id: $courseId) {
    title
  }
}
====== Get All Courses with title =====
query GetAllCourses {
  courses{
    title
  }
}
=============== Using Aliases ========

query GetCoursesById {
  firstcourse:course(id:1) {
    title
  }
  secondcourse:course(id:2) {
    title
  }
}
